
\section*{Resumen de Nomenclaturas}

\subsection*{Nomenclatura Tradicional}
\begin{itemize}
    \item Usa sufijos: -oso (menor valencia), -ico (mayor valencia)
    \item Prefijos: hipo- (la más baja), per- (la más alta)
    \item Ejemplo: \ce{FeCl2} = Cloruro ferroso, \ce{FeCl3} = Cloruro férrico
\end{itemize}

\subsection*{Nomenclatura Stock}
\begin{itemize}
    \item Usa números romanos para indicar la valencia del metal
    \item Ejemplo: \ce{FeCl2} = Cloruro de hierro (II), \ce{FeCl3} = Cloruro de hierro (III)
    \item No se usan números romanos si el metal tiene valencia única
\end{itemize}

\subsection*{Nomenclatura IUPAC (Sistemática)}
\begin{itemize}
    \item Usa prefijos numéricos: mono-, di-, tri-, tetra-, penta-, hexa-, hepta-, etc.
    \item Indica el número de átomos de cada elemento
    \item Ejemplo: \ce{FeCl2} = Dicloruro de hierro, \ce{FeCl3} = Tricloruro de hierro
    \item El prefijo "mono-" generalmente se omite
\end{itemize}
